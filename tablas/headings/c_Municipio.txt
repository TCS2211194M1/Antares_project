CREATE TABLE c_Municipio (
C_MUNICIPIO BIGINT NOT NULL,
C_ESTADO VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
DESCRIPCION VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
FECHA_DE_INICIO_DE_VIGENCIA DATE,
FECHA_DE_FIN_DE_VIGENCIA DATE,
ENTRY_STATUS BIGINT NOT NULL,
CREATE_DATE DATE,
CREATE_BY BIGINT NOT NULL,
UPDATE_DATE DATE,
UPDATE_BY BIGINT NOT NULL,
PRIMARY KEY (C_MUNICIPIO)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
