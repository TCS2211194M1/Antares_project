CREATE TABLE c_Periodicidad (
C_PERIODICIDAD BIGINT NOT NULL,
DESCRIPCION VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
VALOR_DIARIO FLOAT NOT NULL,
VALOR_SEMANAL FLOAT NOT NULL,
VALOR_QUINCENAL FLOAT NOT NULL,
VALOR_MENSUAL FLOAT NOT NULL,
VALOR_BIMESTRAL FLOAT NOT NULL,
VALOR_TRIMESTRAL FLOAT NOT NULL,
VALOR_CUATRIMESTRAL FLOAT NOT NULL,
VALOR_SEMESTRAL FLOAT NOT NULL,
VALOR_ANUAL FLOAT NOT NULL,
VALOR_CATORCENAL FLOAT NOT NULL,
FECHA_INICIO_DE_VIGENCIA DATE,
FECHA_FIN_DE_VIGENCIA DATE,
ENTRY_STATUS BIGINT NOT NULL,
CREATE_DATE DATE,
CREATE_BY BIGINT NOT NULL,
UPDATE_DATE DATE,
UPDATE_BY BIGINT NOT NULL,
PRIMARY KEY (C_PERIODICIDAD)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
