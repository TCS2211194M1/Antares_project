CREATE TABLE t_taxid (
T_TAXID BIGINT NOT NULL,
RFC VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
ADDRESS VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
C_PAIS VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
C_ESTADO VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
C_MUNICIPIO BIGINT NOT NULL,
C_LOCALIDAD BIGINT NOT NULL,
C_CODIGOPOSTAL BIGINT NOT NULL,
C_COLONIA BIGINT NOT NULL,
C_REGIMENFISCAL BIGINT NOT NULL,
C_USOCFDI VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
ENTRY_STATUS BIGINT NOT NULL,
CREATE_DATE DATE,
CREATE_BY BIGINT NOT NULL,
UPDATE_DATE DATE,
UPDATE_BY BIGINT NOT NULL,
PRIMARY KEY (T_TAXID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
