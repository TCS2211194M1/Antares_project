CREATE TABLE t_storage (
T_STORAGE VARCHAR(1000),
DEVICE_NAME VARCHAR(1000),
INTANCE_ATTACHMENT_POINT VARCHAR(1000),
VOLUME_SIZE INT,
VOLUME_TYPE VARCHAR(1000),
IOPS INT,
THROUGHPUT FLOAT,
ENCRYPTED VARCHAR(1000),
DELETE_ON_TERMINATION VARCHAR(1000),
INSTANCE VARCHAR(1000),
ENTRY_STATUS INT,
CREATE_DATE DATE,
CREATE_BY INT,
UPDATE_DATE DATE,
UPDATE_BY INT
);
