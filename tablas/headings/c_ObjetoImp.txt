CREATE TABLE c_ObjetoImp (
C_OBJETOIMP BIGINT NOT NULL,
DESCRIPCION VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
C_IMPUESTO BIGINT NOT NULL,
FECHA_INICIO_DE_VIGENCIA DATE,
FECHA_FIN_DE_VIGENCIA DATE,
ENTRY_STATUS BIGINT NOT NULL,
CREATE_DATE DATE,
CREATE_BY BIGINT NOT NULL,
UPDATE_DATE DATE,
UPDATE_BY BIGINT NOT NULL,
PRIMARY KEY (C_OBJETOIMP)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
