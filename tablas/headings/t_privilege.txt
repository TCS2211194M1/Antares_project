CREATE TABLE t_privilege (
T_PRIVILEGE BIGINT NOT NULL,
SHORT_DESCRIPTION VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
GRANT_TABLE_COLUMN VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
CONTEXT VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
ENTRY_STATUS BIGINT NOT NULL,
CREATE_DATE DATE,
CREATE_BY BIGINT NOT NULL,
UPDATE_DATE DATE,
UPDATE_BY BIGINT NOT NULL,
PRIMARY KEY (T_PRIVILEGE)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
