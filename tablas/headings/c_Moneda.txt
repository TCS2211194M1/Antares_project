CREATE TABLE c_Moneda (
C_MONEDA VARCHAR(1000),
DESCRIPCION VARCHAR(1000),
CURRENCY_SYMBOL VARCHAR(1000),
DECIMALES INT,
PORCENTAJE_VARIACION VARCHAR(1000),
FECHA_INICIO_DE_VIGENCIA TIMESTAMP,
FECHA_FIN_DE_VIGENCIA TIMESTAMP,
ENTRY_STATUS INT,
CREATE_DATE TIMESTAMP,
CREATE_BY INT,
UPDATE_DATE TIMESTAMP,
UPDATE_BY INT
);
