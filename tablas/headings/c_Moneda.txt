CREATE TABLE c_Moneda (
C_MONEDA VARCHAR(1000),
DESCRIPCION VARCHAR(1000),
CURRENCY_SYMBOL VARCHAR(1000),
DECIMALES INT,
PORCENTAJE_VARIACION VARCHAR(1000),
FECHA_INICIO_DE_VIGENCIA DATE,
FECHA_FIN_DE_VIGENCIA DATE,
ENTRY_STATUS INT,
CREATE_DATE DATE,
CREATE_BY INT,
UPDATE_DATE DATE,
UPDATE_BY INT
);
