CREATE TABLE c_Moneda (
C_MONEDA BIGINT NOT NULL,
SHORT_DESCRIPTION VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
LONG_DESCRIPTION VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
CURRENCY_SYMBOL VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
DECIMALES BIGINT NOT NULL,
PORCENTAJE_VARIACION VARCHAR(1000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
FECHA_INICIO_DE_VIGENCIA DATE,
FECHA_FIN_DE_VIGENCIA DATE,
ENTRY_STATUS BIGINT NOT NULL,
CREATE_DATE DATE,
CREATE_BY BIGINT NOT NULL,
UPDATE_DATE DATE,
UPDATE_BY BIGINT NOT NULL,
PRIMARY KEY (C_MONEDA)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
