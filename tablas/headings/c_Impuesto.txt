CREATE TABLE c_Impuesto (
C_IMPUESTO INT,
DESCRIPCION VARCHAR(1000),
RETENCION VARCHAR(1000),
TRASLADO VARCHAR(1000),
LOCAL_O_FEDERAL VARCHAR(1000),
FECHA_INICIO_DE_VIGENCIA TIMESTAMP,
FECHA_FIN_DE_VIGENCIA TIMESTAMP,
ENTRY_STATUS INT,
CREATE_DATE TIMESTAMP,
CREATE_BY INT,
UPDATE_DATE TIMESTAMP,
UPDATE_BY INT
);
