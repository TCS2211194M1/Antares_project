CREATE TABLE c_Impuesto (
C_IMPUESTO INT,
DESCRIPCION VARCHAR(1000),
C_TIPOFACTOR VARCHAR(1000),
C_TASA FLOAT,
RETENCION VARCHAR(1000),
TRASLADO VARCHAR(1000),
LOCAL_O_FEDERAL VARCHAR(1000),
FECHA_INICIO_DE_VIGENCIA DATE,
FECHA_FIN_DE_VIGENCIA DATE,
ENTRY_STATUS INT,
CREATE_DATE DATE,
CREATE_BY INT,
UPDATE_DATE DATE,
UPDATE_BY INT
);
